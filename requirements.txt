# 注意：本项目兼容新版本的torchaudio API
torchaudio>=0.12.0

# SoundFile - torchaudio的音频后端（推荐）

soundfile>=0.12.1

onnxruntime>=1.16.1

# Silero VAD - 语音活动检测
silero-vad>=5.0.0

# YAML配置文件支持
